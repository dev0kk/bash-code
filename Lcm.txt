#!/bin/bash

echo "Enter first number:"
read a

echo "Enter second number:"
read b

# Save original values
x=$a
y=$b

# Find GCD (Euclidean Algorithm)
while [ $y -ne 0 ]
do
    temp=$y
    y=$((x % y))
    x=$temp
done

gcd=$x

# Calculate LCM
lcm=$(( (a * b) / gcd ))

echo "LCM = $lcm"
