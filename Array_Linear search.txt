#!/bin/bash

echo "Enter number of elements:"
read n

echo "Enter array elements:"
for ((i=0; i<n; i++))
do
    read arr[i]
done

echo "Enter element to search:"
read key

found=0

for i in "${arr[@]}"
do
    if [ $i -eq $key ]; then
        found=1
        break
    fi
done

if [ $found -eq 1 ]; then
    echo "Element found"
else
    echo "Element not found"
fi
